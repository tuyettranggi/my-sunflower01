HTML:

example01

<div class="boxSizing01">
<div class="split">This div occupies the left half.</div>
<div class="split">This div occupies the right half.</div>
</div>

example02

<div class="boxSizing02">
<div class="boxes">This div occupies the left half.</div>
<div class="boxes">This div occupies the right half.</div>
</div>

CSS:

example01

div.boxSizing01 {
	width:35em;
	border: 10px solid #900;
	height: 40px;
}
div.boxSizing01 div.split {
	box-sizing: border-box;
	-o-box-sizing: border-box;
	-icab-box-sizing: border-box;
	-khtml-box-sizing: border-box;
	-moz-box-sizing: border-box;
	-webkit-box-sizing: border-box;
	width:50%;
	height: 40px;
	text-align: center;
	border: 10px silver ridge;
	padding: 2px;
	float:left;
}

example02

div.boxSizing02 {
	width: 300px;
	border: 10px solid #ddccb5;
	height: 60px;
}
div.boxSizing02 .boxes {
	box-sizing: border-box;
	width:50%;
	height: 60px;
	text-align: center;
	border: 5px solid #897048;
	padding: 2px;
	float:left;
}
