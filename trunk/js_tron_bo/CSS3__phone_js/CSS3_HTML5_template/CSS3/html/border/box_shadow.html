p.boxShadow {
    background: #ffff33;
    -moz-box-shadow: 2px 4px 6px rgba(0, 0, 0, 0.6);
    -webkit-box-shadow: 2px 4px 6px rgba(0,0,0,0.6);
    box-shadow: 2px 4px 6px rgba(0,0,0,0.6);
    margin-bottom: 10px;
    padding: 10px;
}
