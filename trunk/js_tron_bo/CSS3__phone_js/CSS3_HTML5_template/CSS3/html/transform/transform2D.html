CSS:

p.transformStyle01 {
    background: #ffff33;
    -moz-transform: rotate(-3deg);
    -webkit-transform: rotate(-3deg);
    -o-transform: rotate(-3deg);
    margin: 20px 0;
    padding: 10px;
}

p.transformStyle02 {
	background: #ffff33;
	-moz-transform: skew(-7deg,-3deg);
	-webkit-transform: skew(-7deg,-3deg);
	-o-transform: skew(-7deg,-3deg);
	margin: 20px 0;
	padding: 10px;
}

p.transformStyle03 {
	background: #ffff33;
	border: 1px solid black;
	margin: 110px 0 90px 60px;
	height: 50px;
	transform: scale(2,4);
	-moz-transform: scale(2,4);
	-webkit-transform: scale(2,4);
	-o-transform: scale(2,4);
	width: 100px;
}

	background: #ffff33;
	border: 1px solid black;
	height: 50px;
	transform: matrix(0.866,0.5,-0.5,0.866,0,0);
	-moz-transform: matrix(0.866,0.5,-0.5,0.866,0,0);

	-o-transform: matrix(0.866,0.5,-0.5,0.866,0,0);

}
